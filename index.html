@property declaration

CSS.registerProperty({
  name: '--colorPrimary',
  syntax: '<color>',
  initialValue: 'magenta',
  inherits: false
});

@property Typed CSS

.card {
  background-color: var(--colorPrimary); /* magenta */
}

.highlight-card {
  --colorPrimary: yellow;
  background-color: var(--colorPrimary); /* yellow */
}

.another-card {
  --colorPrimary: 23;
  background-color: var(--colorPrimary); /* magenta */
}
 
@property --gradPoint {
  syntax: '';
  inherits: false;
  initial-value: 40%;
}

.post {
  background: 
    linear-gradient(var(--color1) var(--gradPoint), 
                    var(--color2) calc(var(--gradPoint) + 20%));
  transition: --gradPoint 0.5s, filter 0.8s;
}

.post:hover, .post:focus {
  --gradPoint: 100%;
}
